üì° Connect to MQTT Broker

#include <WiFi.h>
#include <PubSubClient.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
const char* mqtt_server = "192.168.1.100";  // Your broker IP
const int mqtt_port = 1883;

WiFiClient espClient;
PubSubClient client(espClient);

void setup() {
  Serial.begin(115200);
  
  // Connect to WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("WiFi connected");
  
  // Setup MQTT
  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(mqttCallback);
  
  // Connect to MQTT
  reconnect();
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Connecting to MQTT...");
    String clientId = "ESP32-" + String(ESP.getEfuseMac(), HEX);
    
    if (client.connect(clientId.c_str())) {
      Serial.println("connected");
      // Subscribe to GPIO commands
      String topic = "device/" + clientId + "/gpio/set";
      client.subscribe(topic.c_str());
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      delay(5000);
    }
  }
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();
}

üì§ Publish Sensor Data (Telemetry)

#include <ArduinoJson.h>
#include <DHT.h>

#define DHTPIN 4
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

void publishTelemetry() {
  float temperature = dht.readTemperature();
  float humidity = dht.readHumidity();
  
  if (isnan(temperature) || isnan(humidity)) {
    Serial.println("Failed to read from DHT sensor!");
    return;
  }
  
  // Create JSON payload
  StaticJsonDocument<256> doc;
  doc["tC"] = temperature;
  doc["rh"] = humidity;
  doc["heap"] = ESP.getFreeHeap();
  doc["rssi"] = WiFi.RSSI();
  doc["uptime"] = millis() / 1000;
  
  String payload;
  serializeJson(doc, payload);
  
  // Publish to telemetry topic
  String deviceId = "ESP32-" + String(ESP.getEfuseMac(), HEX);
  String topic = "device/" + deviceId + "/telemetry";
  client.publish(topic.c_str(), payload.c_str());
  
  Serial.println("Telemetry sent: " + payload);
}

void setup() {
  // ... WiFi and MQTT setup ...
  dht.begin();
}

void loop() {
  client.loop();
  
  // Send telemetry every 5 seconds
  static unsigned long lastPublish = 0;
  if (millis() - lastPublish > 5000) {
    publishTelemetry();
    lastPublish = millis();
  }
}

üì• Receive GPIO Commands

#include <ArduinoJson.h>

// Define GPIO pins
#define GPIO1 5
#define GPIO2 6
#define GPIO3 7
#define GPIO4 8

void setup() {
  // ... WiFi and MQTT setup ...
  
  // Configure GPIO pins as outputs
  pinMode(GPIO1, OUTPUT);
  pinMode(GPIO2, OUTPUT);
  pinMode(GPIO3, OUTPUT);
  pinMode(GPIO4, OUTPUT);
}

void mqttCallback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Message received on topic: ");
  Serial.println(topic);
  
  // Parse JSON payload
  StaticJsonDocument<256> doc;
  DeserializationError error = deserializeJson(doc, payload, length);
  
  if (error) {
    Serial.println("JSON parsing failed!");
    return;
  }
  
  // Check if it's a GPIO command
  if (doc.containsKey("type") && doc["type"] == "gpio") {
    int pin = doc["pin"];
    bool state = doc["state"];
    
    // Map GPIO number to actual pin
    int actualPin;
    switch(pin) {
      case 1: actualPin = GPIO1; break;
      case 2: actualPin = GPIO2; break;
      case 3: actualPin = GPIO3; break;
      case 4: actualPin = GPIO4; break;
      default: 
        Serial.println("Invalid GPIO pin!");
        return;
    }
    
    // Set GPIO state
    digitalWrite(actualPin, state ? HIGH : LOW);
    Serial.printf("GPIO %d set to %s\\n", pin, state ? "ON" : "OFF");
  }
}

üîå Complete Actuator Example

#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

// WiFi credentials
const char* ssid = "YOUR_WIFI";
const char* password = "YOUR_PASSWORD";
const char* mqtt_server = "192.168.1.100";

WiFiClient espClient;
PubSubClient client(espClient);

// GPIO Pins
const int gpioPins[] = {5, 6, 7, 8, 9, 10, 21, 38};
String deviceId;

void setup() {
  Serial.begin(115200);
  
  // Initialize GPIOs
  for (int i = 0; i < 8; i++) {
    pinMode(gpioPins[i], OUTPUT);
    digitalWrite(gpioPins[i], LOW);
  }
  
  // Connect WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  deviceId = "ACTUATOR-" + String((uint32_t)ESP.getEfuseMac(), HEX);
  
  // Setup MQTT
  client.setServer(mqtt_server, 1883);
  client.setCallback(mqttCallback);
  reconnect();
}

void loop() {
  if (!client.connected()) reconnect();
  client.loop();
  
  // Publish status every 10 seconds
  static unsigned long lastStatus = 0;
  if (millis() - lastStatus > 10000) {
    publishStatus();
    lastStatus = millis();
  }
}

void reconnect() {
  while (!client.connected()) {
    if (client.connect(deviceId.c_str())) {
      String topic = "device/" + deviceId + "/gpio/set";
      client.subscribe(topic.c_str());
    } else {
      delay(5000);
    }
  }
}

void mqttCallback(char* topic, byte* payload, unsigned int length) {
  StaticJsonDocument<256> doc;
  deserializeJson(doc, payload, length);
  
  if (doc["type"] == "gpio") {
    int pin = doc["pin"];
    bool state = doc["state"];
    
    if (pin >= 1 && pin <= 8) {
      digitalWrite(gpioPins[pin-1], state ? HIGH : LOW);
    }
  }
}

void publishStatus() {
  StaticJsonDocument<256> doc;
  doc["heap"] = ESP.getFreeHeap();
  doc["rssi"] = WiFi.RSSI();
  doc["uptime"] = millis() / 1000;
  
  String payload;
  serializeJson(doc, payload);
  client.publish(("device/" + deviceId + "/telemetry").c_str(), payload.c_str());
}

üêç Python Client Example

Basic MQTT Client

import paho.mqtt.client as mqtt
import json

def on_connect(client, userdata, flags, rc):
    print(f"Connected with result code {rc}")
    # Subscribe to all device telemetry
    client.subscribe("device/+/telemetry")

def on_message(client, userdata, msg):
    print(f"Topic: {msg.topic}")
    payload = json.loads(msg.payload.decode())
    print(f"Data: {payload}")

client = mqtt.Client()
client.on_connect = on_connect
client.on_message = on_message

client.connect("localhost", 1883, 60)

# Control GPIO
def control_gpio(device_id, gpio_pin, state):
    topic = f"device/{device_id}/gpio/set"
    payload = json.dumps({"type": "gpio", "pin": gpio_pin, "state": state})
    client.publish(topic, payload)

# Example: Turn ON GPIO 1 on device
control_gpio("ACTUATOR-4e5c", 1, True)

client.loop_forever()
